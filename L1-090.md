#include <stdio.h>
int main(){
    int num1 = 0, num2 = 0;
    int ret = scanf("%d %d", &num1, &num2);
    if(ret != 2){
        return 1;
    }
    int nums[4] = {0};
    int sum = num1 + num2;
    nums[0] = sum - 16;
    nums[1] = sum - 3;
    nums[2] = sum - 1;
    nums[3] = sum;
    for(int i = 0;i < 4;i++){
        printf("%d\n",nums[i]);
    }
    return 0;
}
