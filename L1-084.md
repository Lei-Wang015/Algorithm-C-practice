int fact(int a);
int main() {
    int a = 0, b = 0;
    int ret = scanf("%d %d", &a, &b);
    if(ret != 2){
        return 1;
    }
    int sum = a + b;
    printf("%d\n", fact(sum));
    return 0;
}

int fact(int a){
    int b = 1;
    for(int i = 1; i <= a;i++){
        b *= i;
    }
    return b;
}
