            return 1;
        }
        int sumA = 0, sumB = 0;
        int tempA = a,tempB = b;
        while(tempA != 0){
            sumA += tempA % 10;
            tempA /= 10;
        }
        while(tempB != 0){
            sumB += tempB % 10;
            tempB /= 10;
        }
        if(0 == a % sumB && b % sumA != 0){
            printf("A\n");
        }
        else if(0 == b % sumA && a % sumB != 0){
            printf("B\n");
        }
        else{
            if(a > b){
                printf("A\n");
            }
            else{
                printf("B\n");
            }
        }
    }
    return 0;
}
