```C
#include <stdio.h>
int main() {
    int N = 0, M= 0, K = 0;
    char X[10];
    int ret = scanf("%d %s %d %d", &N, X, &M, &K);
    if(ret != 4){
        return 1;
    }
    if(K == N){
        printf("mei you mai %s de\n",X);
    }else if(K == M){
        printf("kan dao le mai %s de\n", X);
    }
    else{
        printf("wang le zhao mai %s de\n", X);
    }
    return 0;
}
```
- 输入是按顺序读取的，读取字符串不需要取址符，因为字符串本质是数组；
