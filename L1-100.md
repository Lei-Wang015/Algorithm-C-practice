```C
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int people = 0, skill = 0, sum = 0;
    int ret = scanf("%d %d", &people, &skill);
    if(ret != 2){
        return 1;
    }
    int *nums = (int*)malloc(sizeof(int) * skill);
    for(int i = 0;i < skill;i++){
        int ret1 = scanf("%d", &nums[i]);
        if(ret1 != 1){
            return 1;
        }
        sum += nums[i];
    }
    int allround = 0;
    allround = sum - people * (skill - 1);
    //处理allround为负数的情况
    if(allround < 0){
        allround = 0;
    }
    printf("%d\n", allround);
    return 0;
}
```
